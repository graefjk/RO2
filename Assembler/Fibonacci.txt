//Assembler for fibonacci sequence.
//Starting from (0, 1) and running for 10 steps.

LOAD x0, 0
LOAD x1, 1
LOAD x2, 10 //amount of steps
LOAD x3, 0 //counter

//While counter < steps, run fib-loop.
COMPARE x3, x2
JUMPC 8 //Jump to Fib-loop
JUMP 15 //Jump to End

//Fib-loop:
LOAD x4, 0
ADD x4, x0
ADD x4, x1
LOAD x0, x1
LOAD x1, x4
ADD x3, 1
JUMP 5

//End:
LOAD x0, x0
